function createLoadSheets() {
  let totalSheets = ss.getNumSheets();
  if (totalSheets > 1) ss.moveActiveSheet(totalSheets);

  for (let i in json) {
    const jsonItem = json[i];
    let tableColumnNames = Object.keys(jsonItem[0]);
    let rows = [];
    let order = null;

    // --- 🔧 Definición de orden por hoja ---
    switch (sheetNames[i]) {
      case "cards":
        order = ["lessonid", "classroomids", "period", "weeks", "terms", "days", "lessons_per_day"];
        break;
      case "lessons":
        order = ["id", "classids", "subjectid", "periodspercard", "periodsperweek", "teacherids", "classroomids", "groupids", "seminargroup", "termsdefid", "weeksdefid", "daysdefid", "capacity", "partner_id"];
        break;
      case "groups":
        order = ["id", "name", "classid", "studentids", "entireclass", "divisiontag"];
        break;
      case "classes":
        order = ["id", "name", "short", "teacherid", "classroomids", "grade", "partner_id"];
        break;
      case "grades":
        order = ["name", "short", "grade"];
        break;
      case "subjects":
        order = ["id", "name", "short", "partner_id"];
        break;
      case "teachers":
        order = ["id", "firstname", "lastname", "name", "short", "gender", "color", "email", "mobile", "partner_id"];
        break;
      case "buildings":
        order = ["id", "name"];
        break;
      case "periods":
        order = ["name", "short", "period", "starttime", "endtime"];
        break;
      case "daysdefs":
        order = ["days", "short", "name", "id"];
        break;
    }

    // --- 🧩 Aplicar reordenamiento ---
    if (order) {
      const existingCols = order.filter(c => tableColumnNames.includes(c));
      if (existingCols.length >= order.length - 1) {
        rows = jsonItem.map(obj => order.map(k => obj[k] || ""));
        tableColumnNames = order;
        Logger.log(`[createLoadSheets] ${sheetNames[i]} reordenada correctamente.`);
      } else {
        Logger.log(`[createLoadSheets] Falta(n) columna(s) en ${sheetNames[i]}: ${order.filter(c => !tableColumnNames.includes(c)).join(", ")}`);
        rows = jsonItem.map(obj => Object.values(obj));
      }
    } else {
      rows = jsonItem.map(obj => Object.values(obj));
    }

    // --- Invertir si es days ---
    if (sheetNames[i] === sheetDaysName) {
      tableColumnNames.reverse();
      rows = rows.map(r => r.reverse());
    }

    // --- Generar correos automáticos para teachers ---
    if (sheetNames[i] === sheetTeachersName) {
      rows.forEach(r => r[7] = `${r[4]}${emailDomain}`);
    }

    // --- Construir cuerpo ---
    const body = [tableColumnNames, ...rows];
    const name = sheetNames[i];
    let sheet = ss.getSheetByName(name);
    if (!sheet) {
      ss.insertSheet(name);
      sheet = ss.getSheetByName(name);
    }

    if (Array.isArray(body) && body.length > 0) {
      sheet.getRange(1, 1, body.length, body[0].length).setValues(body);
    } else {
      Logger.log(`[createLoadSheets] ${name}: body vacío; no se escribe.`);
    }

    // --- Estilo visual ---
    sheet.getRange(1, 1, 1, body[0].length)
      .setBackground(cellColor)
      .setFontColor(fontColor)
      .setFontWeight("bold");

    if (border) {
      sheet.getRange(1, 1, body.length, body[0].length)
        .setBorder(false, false, true, true, false, false, borderColor, SpreadsheetApp.BorderStyle.SOLID);
    }

    Logger.log(`[createLoadSheets] ${name} cargada correctamente.`);
  }

  Logger.log("Sheets Created & Loaded Ready ✅");
}

function cardGetLessonPerDay() {
  // Asegura hoja Cards
  const cards = ss.getSheetByName(sheetCardsName) || ss.insertSheet(sheetCardsName);

  // Datos de la hoja principal
  const lastRowMain = Math.max(sheetMain.getLastRow() - 1, 0);
  const lastColMain = sheetMain.getLastColumn();

  // Si no hay datos, no intentes escribir filas 0
  if (lastRowMain <= 0 || lastColMain <= 0) {
    Logger.log("[cards] Main sin datos. Se omite escritura.");
    // (opcional) Limpia columna G desde la fila 2
    // cards.getRange(2, 7, cards.getMaxRows() - 1, 1).clearContent();
    return;
  }

  const horary = sheetMain.getRange(2, 1, lastRowMain, lastColMain).getDisplayValues();

  let count = 0;
  const totalPeriod = [];

  horary.forEach((items) => {
    horary.forEach((item) => {
      // Mismo día
      if (items[mainColDay] == item[mainColDay]) {
        // Misma materia, docente y clase
        if (
          items[mainColSubject] == item[mainColSubject] &&
          items[mainColTeacher] == item[mainColTeacher] &&
          items[mainColClass] == item[mainColClass]
        ) {
          count += 1;
        }
      }
    });

    // Guarda el conteo (capado a 10 como en tu lógica original)
    totalPeriod.push([String(Math.min(count, 10))]);
    count = 0;
  });

  // Cabecera
  cards.getRange(1, 7).setBackground(cellColor).setFontColor(fontColor).setFontWeight("bold").setValue("lessons_per_day");

  // Evita setValues con 0 filas
  if (!totalPeriod.length) {
    Logger.log("[cards] Sin resultados para escribir en Cards.");
    return;
  }

  cards.getRange(2, 7, totalPeriod.length, 1).setValues(totalPeriod);
}

function convertNumberToHour_(num) {
  let hours = Math.floor(num / 60);
  let minutes = num % 60;
  return hours + ":" + minutes;
}

function convertHourToMinutes_(timeInHour) {
  let timeParts = timeInHour.split(":");
  return Number(timeParts[0]) * 60 + Number(timeParts[1]);
}

// Get Time
function getTime_(timeNow) {
  let hour = timeNow.getHours();
  let minute = timeNow.getMinutes();
  let seconds = timeNow.getSeconds();

  if (hour <= 9) { hour = "0" + hour; }
  if (minute <= 9) { minute = "0" + minute; }
  if (seconds <= 9) { seconds = "0" + seconds; }

  let time = `${hour}:${minute}`;
  return time;
}
